SELECT sys.Name0 AS [Device],
    dev.Device_Name0 AS [PnP Name],
    dev.Class_Guid0  AS [Class GUID],
    dev.PNP_Device_ID0,
    dev.Status0
FROM v_R_System AS sys
JOIN v_GS_PNP_DEVICE AS dev
    ON dev.ResourceID = sys.ResourceID
WHERE   dev.Class_Guid0 = '{F01A9D53-3FF6-48D2-9F97-C8A7004BE10C}'
ORDER BY sys.Name0;
